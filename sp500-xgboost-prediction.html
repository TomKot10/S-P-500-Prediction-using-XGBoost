# PROJEKT ZALICZENIOWY: Predykcja wzrostów S&P 500 przy użyciu XGBoost
# Założenie projektu: Stworzenie narzędzia do predykcji wzrostu cen akcji na podstawie cen zamknięcia, wolumenu, wskaźnika RSI,SMA, indeksu giełdy, zmienności oraz zwrotów tygodniowych.
# Model będzie codziennie aktualizowany, na podstawie danych poda wynik prawodpobieństwa 10 spółek z najwyższym prawdopodieństwem osiągnięcia zysku, na następnej sesji giełdowej.
# SKŁAD GRUPY: Oskar Kanigowski, Piotr Łukowski, Tomasz Kotliński

# Odpowiedzialność w projekcie
# Oskar Kanigowski : Punkt 1 (znalezienie tematu pracy, instelacja potrzebnych bibliotek, wczytanie danych oraz naprawda pobranych plików oraz filtracja daty)
# Piotr Łukowski : Punkt 2 i 3 (Znalezienie potrzebnych wskaznikow do obliczen oraz ich obliczenie i zaimplementowanie do data frame'a oraz ustalenie warunków wzrostu czy urośnie akcja na nastepny dzien)
# Tomasz Kotliński: Punkt 4,5 i 6 ( Konfiguracja modelu, prognoza na kolejna sesje oraz backtest)
# Grupowo ocenilibyśmy wkład na 33% dla kazdego czlonka zespołu

import kagglehub   # potrzebna do pobrania bezposrednio po API strony
import pandas as pd  # do filtracji i sortowania naszych danych
import pandas_ta as ta  # potrzebna do obliczenia RSI oraz SMA
import xgboost as xgb   # potrzebna do naszego cuda nauczania maszynowego !
import matplotlib.pyplot as plt  # potrzebna do wizualizacji wykresów
import numpy as np   # potrzebna do obliczeń
import os          # potrzebna do pobrania plików z kaggla

# 1. Pobranie i wczytanie danych

# Fragment kodu pobrany z strony Kaggle do automatycznego pobierania danych po API
path = kagglehub.dataset_download("joebeachcapital/s-and-p500-index-stocks-daily-updated")

csv_file = None
for root, dirs, files in os.walk(path):
    for file in files:
        if file.endswith("sp500_data.csv"):
            csv_file = os.path.join(root, file)
            break


# Po pobraniu manualnie okazalo sie że wartości z wiersza 2 weszły na nagłówki
# Zostawiliśmy 8 kolumn
df = pd.read_csv(csv_file, header=None).iloc[:, :8]
# Ręcznie zmieniliśmy nazwe kolumn
df.columns = ["Date", "Ticker", "Adj Close", "Close", "High", "Low", "Open", "Volume"]
# Usuneliśmy 1 wiersz z błędymi doczepionym fragmentem do slowa Volume
df = df.iloc[1:].reset_index(drop=True)
# Zmiana na wartości numeryczne
num_cols = ["Adj Close", "Close", "High", "Low", "Open", "Volume"]
df[num_cols] = df[num_cols].apply(pd.to_numeric, errors="coerce")
# Zmiana na Date
df['Date'] = pd.to_datetime(df['Date'])


#Filtorwanie danych po 2023 funkcją copy
df = df[df['Date'] >= '2023-01-01'].copy()

# Sprawdzenie czy mamy wystarczająco próbek
liczba = (df['Date'] >= '2023-01-01').sum()

print("Liczba obserwacji po 2023:", liczba)
# Stan na 16.01.2016 (434745)
# 2. Obliczenia na podstawie zmiennych

# Obliczenie średniej ceny zamknięcia pozycji dla 50 ostatnich sesji po grupowaniu "Ticker"(Skrót giełdowy) i cenie zamkniecia
df['SMA_50'] = df.groupby('Ticker')['Close'].transform(lambda x: x.rolling(50).mean())
# Wskaznik jak wzroslo lub jak zmalało od ostatniej ceny zamkniecia przy sredniej z tego okresu
df['SMA_50_Dist'] = df['Close'] / df['SMA_50']

df['RSI'] = df.groupby('Ticker')['Close'].transform(lambda x: ta.rsi(x, length=14))
# Wskaznik czy wolumen jest większy niż zwykle na podstawie okresu 20 sejsji giełdowych
df['Rel_Volume'] = df['Volume'] / df.groupby('Ticker')['Volume'].transform(lambda x: x.rolling(20).mean())
# Wskaznik odchylenia standardowego w okresie 14 sesji giełdowych
df['Volatility'] = df.groupby('Ticker')['Close'].transform(lambda x: x.rolling(14).std() / x)
# Wskaznik zysku z ostatnich 5 sesji giełdowych
df['Return_Week'] = df.groupby('Ticker')['Close'].pct_change(5)
# Wskaznik porownwczy Indexu RSI w prównaniu do reszty spółek
df['RSI_Rank'] = df.groupby('Date')['RSI'].rank(pct=True)
# Wskaznik porownwczy Volumenu obrotu w prównaniu do reszty spółek
df['Vol_Rank'] = df.groupby('Date')['Rel_Volume'].rank(pct=True)

# Stworzenie benchmarka do obliczeń tutaj średnia cena zamknięcia całej giełdy
market = df.groupby('Date')['Close'].mean().to_frame('Market_Close')
# Wskaznik zysku calego sektora w ostatniej sesji giełdowej
market['Market_Return'] = market['Market_Close'].pct_change()
# Wskaznik RSI dla cłej giełdy w okresie 14 sesji
market['Market_RSI'] = ta.rsi(market['Market_Close'], length=14)
# Wskaznik średniej ceny zamkniecia dla 5o ostanich sesji dla calej gieldy
market['Market_SMA_50'] = ta.sma(market['Market_Close'], length=50)
## Wskaznik jak wzroslo lub jak zmalało od ostatniej ceny zamkniecia przy sredniej z tego okresu
market['Market_Dist'] = market['Market_Close'] / market['Market_SMA_50']

df = df.merge(market, on='Date', how='left')

# 3. Ustalenie warunków wzrostu/straty

df['Next_Close'] = df.groupby('Ticker')['Close'].shift(-1)
df['Next_Return'] = (df['Next_Close'] - df['Close']) / df['Close']
df['Target'] = (df['Next_Return'] > 0.00001).astype(int)
df = df.dropna()

# Okres testwy od 2024 do najświeższej daty
df_final = df[df['Date'] >= '2024-01-01'].copy()
last_date = df_final['Date'].max()

train_df = df_final[df_final['Date'] < last_date]
future_df = df_final[df_final['Date'] == last_date].copy()
# Wskazniki
features = [
    'RSI', 'Rel_Volume', 'Return_Week', 'Volatility', 'SMA_50_Dist',
    'Market_RSI', 'Market_Dist', 'Market_Return',
    'RSI_Rank', 'Vol_Rank'
]

X_train = train_df[features]
y_train = train_df['Target']


# 4. Konfiguracja Modelu

model = xgb.XGBClassifier(
    n_estimators=150,
    max_depth=6,
    learning_rate=0.05,
    random_state=42,
    eval_metric='logloss'
)
print("""
Skumulowany wynik:

Nasz model wykorzystuje algorytm XGBoost do przewidywania prawdopodobieństwa
wzrostu cen akcji z indeksu S&P 500 na podstawie wskaźników technicznych
i rynkowych. Analiza ważności cech wykazała, że największy wpływ na prognozy
mają trendy rynkowe oraz wskaźnik Market_RSI, co pozwala skutecznie filtrować
spółki o największym potencjale zysku.
""")
model.fit(X_train, y_train)
plt.figure(figsize=(10, 6))
xgb.plot_importance(model, max_num_features=10)
plt.title("Ważność cech")
plt.show()

# 5. Prognoza na kolejną sesje

print(f"\n=== PROGNOZA NA SESJĘ PO: {last_date.date()} ===")
X_future = future_df[features]
future_df['Prob'] = model.predict_proba(X_future)[:, 1]

print(future_df[['Ticker', 'Close', 'Prob']].sort_values('Prob', ascending=False).head(10))

# 6. Weryfikacja backtestem z ostatnich 30 sesji


test_dates = sorted(df_final['Date'].unique())[-30:]
results = []

for d in test_dates:
    day = df_final[df_final['Date'] == d].copy()
    if day.empty: continue
    day['Prob'] = model.predict_proba(day[features])[:, 1]
    top5 = day.sort_values('Prob', ascending=False).head(5)
    results.append(top5['Next_Return'].mean())

results = np.array(results)
cum = (1 + results).cumprod() - 1

print(f"Skumulowany wynik: {cum[-1]*100:.2f}%")
print(f"Średni dzienny wynik zwrotu: {results.mean()*100:.4f}%")
print("""
Wykres skumulowanego zysku przedstawia procentowy przyrost kapitału w czasie,
jaki uzyskałaby strategia inwestycyjna oparta na codziennych typowaniach
modelu XGBoost. Linia ta obrazuje realną skuteczność modelu w okresie testowym,
pokazując, jak codzienne wybory topowych spółek przekładają się na ostateczną
stopę zwrotu.
""")
plt.plot(cum * 100)
plt.title("Backtest: Skumulowany zysk (%)")
plt.grid(True)
plt.show()
